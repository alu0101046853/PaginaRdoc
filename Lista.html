def extract_head
        if @head == nil
                return nil
        elsif @head == @tail
                aux = @head
                @tail = nil
                @head = nil
                @tamano = 0
                return aux
        else
                aux = @head
                @head = aux.next
                @head.prev = nil
                @tamano-=1
                return aux
        end
end

    # File lib/alimentos/lista.rb, line 24
def insertar(valor)
        nodo = Node.new(valor,nil,nil)
        if @tamano == 0
                nodo.next=nil
                nodo.prev=nil
                @head=nodo
                @tail=nodo
                @tamano+=1
        else
                nodo.prev = @tail 
                nodo.next = nil
                @tail.next = nodo 
                @tail = nodo
                @tamano += 1
        end
end
